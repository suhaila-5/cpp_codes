bool prime()
{
    int i=1,x,count=0;
    cout <<"enter your number"<<endl;
    cin>>x;
    while (i<=x)
    {
        if(x%i==0)
        {
            count++;
        }
        i++;
    }
    if(count==2)
    {
        return true ;
    }
    else
    {
        return false;
    }

}
